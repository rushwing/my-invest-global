# Goal

建立 Phase 8b LanceDB RAG 的基础组件：BGE-M3 本地 embedding 包装器（无 API 费用，与
`math-to-go` 共用 `~/.cache/huggingface/` 模型 cache）和 LanceDB 文档 chunk 表

# In Scope

## `engine/rag/embedder.py`

```python
from __future__ import annotations
import numpy as np

_DEFAULT_MODEL = "BAAI/bge-m3"
_DEFAULT_RERANKER = "BAAI/bge-reranker-v2-m3"

class Embedder:
    """Lazy-loaded BGE-M3 dense embedder. Model downloaded once to HF cache."""

    def __init__(self, model_name: str = _DEFAULT_MODEL, device: str = "cpu") -> None:
        self._model_name = model_name
        self._device = device
        self._model = None   # FlagEmbedding.BGEM3FlagModel, imported lazily

    def _get_model(self):
        if self._model is None:
            from FlagEmbedding import BGEM3FlagModel   # lazy import: avoids torch load on tests
            self._model = BGEM3FlagModel(self._model_name, use_fp16=False, device=self._device)
        return self._model

    def embed(self, texts: list[str], batch_size: int = 12) -> np.ndarray:
        """Return dense vectors, shape (len(texts), 1024)."""
        model = self._get_model()
        result = model.encode(texts, batch_size=batch_size, max_length=8192)
        vecs: np.ndarray = result["dense_vecs"]
        return vecs

class Reranker:
    """BGE-reranker-v2-m3 cross-encoder for ANN result re-ranking."""

    def __init__(self, model_name: str = _DEFAULT_RERANKER, device: str = "cpu") -> None:
        from FlagEmbedding import FlagReranker
        self._reranker = FlagReranker(model_name, use_fp16=False, device=device)

    def rerank(self, query: str, passages: list[str]) -> list[float]:
        """Return relevance scores (higher = more relevant) for each passage."""
        pairs = [[query, p] for p in passages]
        return self._reranker.compute_score(pairs)
```

## `engine/rag/schema.py`

```python
from __future__ import annotations
import lancedb
from lancedb.pydantic import LanceModel, Vector

class DocChunk(LanceModel):
    chunk_id: str       # "{doc_id}_{chunk_index:04d}"
    doc_id: str
    title: str
    doc_type: str       # "annual_report" | "research_report" | "announcement" | "news"
    pub_date: str       # ISO date string "YYYY-MM-DD"
    source: str         # "cninfo" | "eastmoney" | "akshare" | "local"
    code: str           # primary stock code; empty string if multi-stock document
    chunk_index: int
    chunk_total: int
    text: str
    vector: Vector(1024)   # BGE-M3 dense embedding

class RAGSchema:
    TABLE_NAME = "doc_chunks"

    @staticmethod
    def create_table(db: lancedb.LanceDBConnection) -> lancedb.table.LanceTable:
        """Idempotent: returns existing table if already created."""
        return db.create_table(RAGSchema.TABLE_NAME, schema=DocChunk, exist_ok=True)
```

# Implementation Notes

- `Embedder` 惰性 import `FlagEmbedding`，避免测试时因缺少 torch/模型文件而失败
- `Vector(1024)` 是 lancedb 的固定维度声明；BGE-M3 输出恒为 1024 维
- `tantivy` 是 LanceDB FTS (full-text search) 的 Rust 后端，实际检索时可选
- `use_fp16=False` on CPU；MPS/CUDA 设备时 fp16 可开启（device 参数控制）
