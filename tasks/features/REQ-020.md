## Summary

实现 `app/dashboard.py` 主入口和 `app/data_loader.py` 数据加载模块，将认证、侧边栏、4 个 Tab 和所有数据源连接为完整应用。包含 watchdog 文件监控和 `@st.cache_data` 缓存策略。

## Acceptance Criteria

- `uv run streamlit run app/dashboard.py` 启动无报错，登录表单正确显示
- 侧边栏编辑持仓并保存 → Tab 1 Treemap 在下一次 st.rerun() 后反映新 market_value（不超过 2 秒）
- 所有数据加载函数使用 `@st.cache_data(ttl=300)` 缓存
- 侧边栏 Save 按钮触发 `st.cache_data.clear()` + `st.rerun()`（MVP 手动刷新，不依赖 watchdog）
- 无数据文件时所有 Tab 显示友好占位符，不抛 KeyError / FileNotFoundError

### compute_portfolio_metrics(df)

```python
def compute_portfolio_metrics(df: pd.DataFrame, macro_state: str) -> dict:
    """不缓存，纯计算。macro_state 接受小写 green/yellow/red（兼容大写，自动 .lower()）。
    df 为空或 total_mv == 0 时返回 elastic_target 但各比例为 0.0。"""
    macro_key = macro_state.lower()
    targets = {"green": 38, "yellow": 33, "red": 20}
    elastic_target = targets.get(macro_key, 33)

    total_mv = df["market_value"].sum() if not df.empty else 0.0
    if total_mv == 0:
        return {
            "total_mv": 0.0,
            "white_mv": 0.0,
            "elastic_mv": 0.0,
            "white_pct": 0.0,
            "elastic_pct": 0.0,
            "elastic_target": elastic_target,
            "df": df,
        }

    white_mv   = df[df["category"] == "白马股"]["market_value"].sum()
    elastic_mv = df[df["category"] == "弹性股"]["market_value"].sum()
    white_pct   = white_mv / total_mv * 100
    elastic_pct = elastic_mv / total_mv * 100
    df = df.copy()
    df["pct"] = df["market_value"] / total_mv * 100
    df["target_pct"] = df.apply(lambda r: elastic_target if r["category"] == "弹性股"
                                else (100 - elastic_target), axis=1)
    df["deviation"] = df["pct"] - df["target_pct"]
    return {
        "total_mv": total_mv,
        "white_mv": white_mv,
        "elastic_mv": elastic_mv,
        "white_pct": white_pct,
        "elastic_pct": elastic_pct,
        "elastic_target": elastic_target,
        "df": df,
    }
```

### load_fund_flow_5d(codes, days=5)

```python
@st.cache_data(ttl=300)
def load_fund_flow_5d(codes: list[str], days: int = 5) -> pd.DataFrame:
    """
    从 DuckDB stock_fund_flow 读近 N 个交易日主力净流入。
    返回 DataFrame: columns=[code, trade_date, main_net_inflow]
    无数据时返回空 DataFrame，不抛异常。
    """
    if not codes:
        return pd.DataFrame(columns=["code", "trade_date", "main_net_inflow"])
    try:
        conn = duckdb.connect(DB_PATH, read_only=True)
        # 参数化 IN 查询：DuckDB 支持 list 参数作为 = ANY($1)
        return conn.execute(
            """
            SELECT code, trade_date, main_net_inflow
            FROM stock_fund_flow
            WHERE code = ANY($1)
              AND trade_date >= CURRENT_DATE - INTERVAL (CAST($2 AS INTEGER) + 2) DAY
            ORDER BY code, trade_date
            """,
            [codes, days],
        ).df()
    except Exception:
        return pd.DataFrame(columns=["code", "trade_date", "main_net_inflow"])
```

### load_market_breadth(days=20)

```python
@st.cache_data(ttl=300)
def load_market_breadth(days: int = 20) -> pd.DataFrame:
    """
    近 N 日每日涨跌家数比。从 stock_quotes 按日期分组聚合。
    返回 DataFrame: columns=[date, up_ratio]  (up_ratio ∈ [0,1])
    """
    try:
        conn = duckdb.connect(DB_PATH, read_only=True)
        return conn.execute("""
            SELECT DATE(quote_time) AS date,
                   SUM(CASE WHEN pct_change > 0 THEN 1 ELSE 0 END)::DOUBLE
                   / NULLIF(COUNT(*), 0) AS up_ratio
            FROM stock_quotes
            WHERE DATE(quote_time) >= CURRENT_DATE - INTERVAL ? DAY
            GROUP BY DATE(quote_time)
            ORDER BY date
        """, [days]).df()
    except Exception:
        return pd.DataFrame(columns=["date", "up_ratio"])
```

### compute_indicators(df)

```python
def compute_indicators(df: pd.DataFrame) -> pd.DataFrame:
    """
    纯计算，不缓存。输入：含 open/high/low/close/volume 列、按 date 升序排列的 DataFrame。
    返回附加列：ma65, bb_upper, bb_mid, bb_lower, macd, macd_signal, macd_hist, rsi14。
    行数不足（< 26）时各指标列填 NaN，不抛异常。
    """
    import pandas_ta as ta
    df = df.copy()
    if len(df) >= 65:
        df["ma65"] = ta.sma(df["close"], length=65)
    else:
        df["ma65"] = float("nan")
    bb = ta.bbands(df["close"], length=20, std=2)
    if bb is not None:
        df["bb_upper"] = bb["BBU_20_2.0"]
        df["bb_mid"]   = bb["BBM_20_2.0"]
        df["bb_lower"] = bb["BBL_20_2.0"]
    else:
        df[["bb_upper", "bb_mid", "bb_lower"]] = float("nan")
    macd = ta.macd(df["close"], fast=12, slow=26, signal=9)
    if macd is not None:
        df["macd"]        = macd["MACD_12_26_9"]
        df["macd_signal"] = macd["MACDs_12_26_9"]
        df["macd_hist"]   = macd["MACDh_12_26_9"]
    else:
        df[["macd", "macd_signal", "macd_hist"]] = float("nan")
    df["rsi14"] = ta.rsi(df["close"], length=14)
    return df
```

### compute_trigger_status(holdings_df, macro_state)

```python
def compute_trigger_status(holdings_df: pd.DataFrame, macro_state: str) -> pd.DataFrame:
    """
    纯计算，不缓存。基于当前持仓和宏观状态计算 7 条再平衡触发器状态。
    返回 DataFrame: columns=[trigger, direction, magnitude, status]
      status: "normal" | "watch" | "triggered"
    holdings_df 为空时所有触发器返回 "normal"。
    """
    macro_key = macro_state.lower()
    targets = {"green": 0.38, "yellow": 0.33, "red": 0.20}
    elastic_target = targets.get(macro_key, 0.33)

    triggers = [
        {"trigger": "云 CapEx 指引上调（季度）",  "direction": "弹性股 ↑", "magnitude": "+7%，上限 40%"},
        {"trigger": "云 CapEx 指引下调 >10%",     "direction": "弹性股 ↓", "magnitude": "-13%，下限 20%"},
        {"trigger": "板块整体回撤 >15%",           "direction": "白马股 ↑", "magnitude": "+7%，上限 75%"},
        {"trigger": "板块突破前高（放量）",         "direction": "弹性股 ↑", "magnitude": "+5%，上限 38%"},
        {"trigger": "单只弹性股亏损 >20%",         "direction": "止损提示", "magnitude": "建议减仓/退出"},
        {"trigger": "单只持仓达目标价 80%",        "direction": "止盈提示", "magnitude": "建议减仓 30–50%"},
        {"trigger": "单只弹性股超 10%",            "direction": "调仓提示", "magnitude": "建议减至 <8%"},
    ]
    rows = []
    for t in triggers:
        status = "normal"
        if not holdings_df.empty:
            total_mv = holdings_df["market_value"].sum()
            if total_mv > 0:
                elastic_pct = (
                    holdings_df[holdings_df["category"] == "弹性股"]["market_value"].sum()
                    / total_mv
                )
                deviation = elastic_pct - elastic_target
                if abs(deviation) > 0.10:
                    status = "triggered"
                elif abs(deviation) > 0.05:
                    status = "watch"
        rows.append({**t, "status": status})
    return pd.DataFrame(rows)
```
